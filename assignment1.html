C Programming Concepts

What is a String?

In C programming, a string is a sequence of characters terminated with a null character '\0'. Strings are actually one-dimensional array of characters terminated by a null character.

Strings are used to store text/characters. For example, "Hello World" is a string of characters.

String Declaration

There are two ways to declare a string in C:

// Method 1: Using character array
char str1[] = "Hello";

// Method 2: Using pointer
char *str2 = "World";

Note: The null character '\0' is automatically added at the end when you initialize a character array with a string literal.

Pointer in C

A pointer is a variable that stores the memory address of another variable. Pointers are powerful features of C that allow you to directly manipulate memory.

Pointer Declaration

data_type *pointer_name;

Where:

  • data_type is the type of data the pointer points to
  • * indicates that the variable is a pointer
  • pointer_name is the name of the pointer variable

Example

#include <stdio.h>

int main() {
    int var = 20;   // Actual variable declaration
    int *ip;        // Pointer variable declaration
    
    ip = &var;      // Store address of var in pointer variable
    
    printf("Address of var variable: %p\n", &var);
    printf("Address stored in ip variable: %p\n", ip);
    printf("Value of *ip variable: %d\n", *ip);
    
    return 0;
}

Pointer Arithmetic

Pointers support arithmetic operations:

  • Increment (++)
  • Decrement (--)
  • Addition (+)
  • Subtraction (-)

Dynamic Variable

A dynamic variable is a variable whose memory is allocated at runtime (during program execution) rather than at compile time. In C, dynamic memory allocation is done using functions like malloc(), calloc(), realloc(), and free() from the stdlib.h library.

Key Functions

  • malloc() - Allocates a block of memory
  • calloc() - Allocates multiple blocks of memory and initializes them to zero
  • realloc() - Reallocates memory to increase or decrease size
  • free() - Frees previously allocated memory

Example

#include <stdio.h>
#include <stdlib.h>

int main() {
    int *ptr;
    int n = 5;
    
    // Dynamically allocate memory using malloc()
    ptr = (int*)malloc(n * sizeof(int));
    
    if (ptr == NULL) {
        printf("Memory not allocated.\n");
        exit(0);
    }
    
    // Initialize the array
    for (int i = 0; i < n; i++) {
        ptr[i] = i + 1;
    }
    
    // Print the array
    for (int i = 0; i < n; i++) {
        printf("%d ", ptr[i]);
    }
    
    // Free the memory
    free(ptr);
    
    return 0;
}

What is a File?

A file is a container in computer storage devices used for storing data. Files are used to preserve information even after the program terminates.

File Operations

Call by Value and Call by Reference

Call by Value

In call by value, a copy of the actual parameter's value is passed to the function. Changes made to the parameter inside the function have no effect on the argument.

#include <stdio.h>

void swap(int x, int y) {
    int temp = x;
    x = y;
    y = temp;
    printf("Inside swap: x = %d, y = %d\n", x, y);
}

int main() {
    int a = 10, b = 20;
    printf("Before swap: a = %d, b = %d\n", a, b);
    swap(a, b);
    printf("After swap: a = %d, b = %d\n", a, b);
    return 0;
}

Output:
Before swap: a = 10, b = 20
Inside swap: x = 20, y = 10
After swap: a = 10, b = 20

Call by Reference

In call by reference, the address of the variable is passed to the function. Changes made to the parameter affect the passed argument.

#include <stdio.h>

void swap(int *x, int *y) {
    int temp = *x;
    *x = *y;
    *y = temp;
    printf("Inside swap: *x = %d, *y = %d\n", *x, *y);
}

int main() {
    int a = 10, b = 20;
    printf("Before swap: a = %d, b = %d\n", a, b);
    swap(&a, &b);
    printf("After swap: a = %d, b = %d\n", a, b);
    return 0;
}

Output:
Before swap: a = 10, b = 20
Inside swap: *x = 20, *y = 10
After swap: a = 20, b = 10

Pointer to Array

A pointer to an array is a pointer that points to the first element of an array. Array name itself acts as a pointer to the first element of the array.

Example

#include <stdio.h>

int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    int *ptr = arr;  // Pointer to array
    
    printf("Array elements using pointer:\n");
    for (int i = 0; i < 5; i++) {
        printf("arr[%d] = %d\n", i, *(ptr + i));
    }
    
    return 0;
}

Pointer to Multidimensional Array

#include <stdio.h>

int main() {
    int arr[2][3] = {{1, 2, 3}, {4, 5, 6}};
    int (*ptr)[3] = arr;  // Pointer to 2D array
    
    printf("2D array elements using pointer:\n");
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            printf("%d ", ptr[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}

String Functions

C provides several built-in string handling functions in the string.h library. Here are four important ones:

1. strlen() - String Length

Returns the length of the string (number of characters before the null character).

#include <stdio.h>
#include <string.h>

int main() {
    char str[] = "Hello World";
    printf("Length of string: %lu\n", strlen(str));
    return 0;
}

Output: Length of string: 11

2. strcpy() - String Copy

Copies the source string to the destination string.

#include <stdio.h>
#include <string.h>

int main() {
    char src[] = "Copy this";
    char dest[20];
    
    strcpy(dest, src);
    printf("Source: %s\n", src);
    printf("Destination: %s\n", dest);
    
    return 0;
}

Output:
Source: Copy this
Destination: Copy this

3. strcat() - String Concatenation

Appends the source string to the destination string.

#include <stdio.h>
#include <string.h>

int main() {
    char str1[20] = "Hello";
    char str2[] = " World";
    
    strcat(str1, str2);
    printf("Concatenated string: %s\n", str1);
    
    return 0;
}

Output: Concatenated string: Hello World

4. strcmp() - String Compare

Compares two strings lexicographically.

#include <stdio.h>
#include <string.h>

int main() {
    char str1[] = "apple";
    char str2[] = "banana";
    
    int result = strcmp(str1, str2);
    
    if (result < 0)
        printf("'%s' comes before '%s'\n", str1, str2);
    else if (result > 0)
        printf("'%s' comes after '%s'\n", str1, str2);
    else
        printf("Strings are equal\n");
    
    return 0;
}

Output: 'apple' comes before 'banana'

Structure vs Union

Both structures and unions are user-defined data types in C, but they have key differences:

Feature Structure (struct) Union (union)
Memory Allocation Each member has its own memory location All members share the same memory location
Size Sum of sizes of all members (plus padding) Size of the largest member
Access All members can be accessed at any time Only one member can be accessed at a time
Initialization All members can be initialized Only the first member can be initialized
Use Case When you need to store and access all members When you need to store only one member at a time

Example

#include <stdio.h>

// Structure example
struct Student {
    char name[50];
    int age;
    float marks;
};

// Union example
union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    printf("Size of struct Student: %lu bytes\n", sizeof(struct Student));
    printf("Size of union Data: %lu bytes\n", sizeof(union Data));
    
    return 0;
}

Nested Structure

A nested structure is a structure within another structure. This allows you to create complex data types by combining simpler ones.

Example

#include <stdio.h>
#include <string.h>

// Structure for Address
struct Address {
    char street[50];
    char city[30];
    char state[30];
    int pin;
};

// Structure for Employee with nested Address structure
struct Employee {
    char name[50];
    int emp_id;
    float salary;
    struct Address address;  // Nested structure
};

int main() {
    // Create an Employee variable
    struct Employee emp1;
    
    // Assign values
    strcpy(emp1.name, "John Doe");
    emp1.emp_id = 101;
    emp1.salary = 50000.0;
    
    // Assign values to nested address structure
    strcpy(emp1.address.street, "123 Main St");
    strcpy(emp1.address.city, "New York");
    strcpy(emp1.address.state, "NY");
    emp1.address.pin = 10001;
    
    // Display information
    printf("Employee Name: %s\n", emp1.name);
    printf("Employee ID: %d\n", emp1.emp_id);
    printf("Salary: %.2f\n", emp1.salary);
    printf("Address: %s, %s, %s - %d\n", 
           emp1.address.street, 
           emp1.address.city, 
           emp1.address.state, 
           emp1.address.pin);
    
    return 0;
}

Structure vs Array

Feature Structure Array
Data Type Can hold different data types Holds elements of same data type
Access Members accessed by name Elements accessed by index
Memory Members have separate memory locations Elements stored in contiguous memory
Size Size is sum of all members (plus padding) Size is (number of elements) Ã— (size of one element)
Initialization Members can be initialized individually All elements can be initialized at once
Use Case When you need to group related data of different types When you need to store multiple items of same type

Example

#include <stdio.h>

// Structure example
struct Point {

    int y;

// Array example
int main() {

        printf("%d ", numbers[i]);
    }
    printf("\n");
    
    return 0;
}

Array of Pointers

An array of pointers is an array where each element is a pointer to a value of a specific data type. This is useful when you need to store multiple addresses or create arrays of strings.

Example with Integers

#include <stdio.h>

int main() {
    int a = 10, b = 20, c = 30;
    int *ptr_arr[3];  // Array of 3 integer pointers
    
    ptr_arr[0] = &a;
    ptr_arr[1] = &b;
    ptr_arr[2] = &c;
    
    for (int i = 0; i < 3; i++) {
        printf("Value at ptr_arr[%d] = %d\n", i, *ptr_arr[i]);
    }
    
    return 0;
}

Example with Strings

#include <stdio.h>

int main() {
    char *names[] = {"Alice", "Bob", "Charlie"};  // Array of string pointers
    
    for (int i = 0; i < 3; i++) {
        printf("Name %d: %s\n", i+1, names[i]);
    }
    
    return 0;
}

Library Functions for File Handling

C provides several functions in stdio.h for file operations. Here are the key functions with examples:

1. fopen() - Open a File

FILE *fopen(const char *filename, const char *mode);

Modes: "r" (read), "w" (write), "a" (append), "r+" (read/write), "w+" (read/write, truncate), "a+" (read/append)

2. fclose() - Close a File

int fclose(FILE *stream);

3. fprintf() - Write Formatted Data to File

int fprintf(FILE *stream, const char *format, ...);

4. fscanf() - Read Formatted Data from File

int fscanf(FILE *stream, const char *format, ...);

5. fgetc() - Read a Character from File

int fgetc(FILE *stream);

6. fputc() - Write a Character to File

int fputc(int char, FILE *stream);

7. fgets() - Read a String from File

char *fgets(char *str, int n, FILE *stream);

8. fputs() - Write a String to File

int fputs(const char *str, FILE *stream);

Example: File Read/Write

#include <stdio.h>

int main() {
    FILE *file;
    char data[100];
    
    // Writing to a file
    file = fopen("example.txt", "w");
    if (file == NULL) {
        printf("Error opening file!\n");
        return 1;
    }
    fprintf(file, "Hello, File Handling in C!\n");
    fclose(file);
    
    // Reading from a file
    file = fopen("example.txt", "r");
    if (file == NULL) {
        printf("Error opening file!\n");
        return 1;
    }
    fgets(data, 100, file);
    printf("Data from file: %s", data);
    fclose(file);
    
    return 0;
}

Key Terms Explained

Pointer

A pointer is a variable that stores the memory address of another variable. Pointers provide direct access to memory locations and are used for dynamic memory allocation, array handling, and function arguments.

Key Concepts:

  • Declaration: int *ptr;
  • Address operator: & (gets address of a variable)
  • Dereference operator: * (accesses value at pointer address)
  • Pointer arithmetic: Adding/subtracting integers to pointers

Structures

A structure is a user-defined data type that allows grouping of variables of different types under a single name. Structures are used to represent records or complex data entities.

Key Features:

  • Defined using struct keyword
  • Members can be of different data types
  • Members accessed using dot (.) operator
  • Supports nesting (structures within structures)

Typedef

The typedef keyword is used to create an alias name for existing data types. It's often used with structures to simplify syntax.

typedef struct {
    char name[50];
    int age;
} Person;

// Now we can use Person directly
Person p1;
strcpy(p1.name, "Alice");
p1.age = 25;

Dynamic Memory Allocation

Dynamic memory allocation is the process of allocating memory at runtime rather than at compile time. This allows programs to request memory when needed and release it when no longer needed.

Functions:

  • malloc() - Allocates memory block
  • calloc() - Allocates and initializes memory block
  • realloc() - Resizes previously allocated block
  • free() - Releases allocated memory

Example:

int *arr = (int*)malloc(5 * sizeof(int));  // Array of 5 integers
if (arr != NULL) {
    // Use the array
    free(arr);  // Release memory when done
}

Student Structure Program

This program demonstrates the use of structures to store student information and calculate the average of total marks.

#include <stdio.h>
#include <string.h>

// Define Student structure
struct Student {
    char name[50];
    int age;
    float total_marks;
};

int main() {
    struct Student students[2];
    float sum = 0, average;
    
    // Input data for two students
    for (int i = 0; i < 2; i++) {
        printf("\nEnter details for student %d:\n", i+1);
        
        printf("Name: ");
        fgets(students[i].name, 50, stdin);
        students[i].name[strcspn(students[i].name, "\n")] = '\0';  // Remove newline
        
        printf("Age: ");
        scanf("%d", &students[i].age);
        
        printf("Total Marks: ");
        scanf("%f", &students[i].total_marks);
        
        getchar();  // To consume the newline left by scanf
        
        sum += students[i].total_marks;
    }
    
    // Calculate average
    average = sum / 2;
    
    // Display student information
    printf("\nStudent Information:\n");
    for (int i = 0; i < 2; i++) {
        printf("\nStudent %d:\n", i+1);
        printf("Name: %s\n", students[i].name);
        printf("Age: %d\n", students[i].age);
        printf("Total Marks: %.2f\n", students[i].total_marks);
    }
    
    // Display average
    printf("\nAverage of Total Marks: %.2f\n", average);
    
    return 0;
}

Program Explanation

  1. We define a Student structure with three members: name, age, and total_marks.
  2. In main(), we create an array of two Student structures.
  3. We use a loop to input data for each student, using fgets() for the name to handle spaces.
  4. We calculate the sum of marks while inputting data.
  5. After input, we calculate the average of the total marks.
  6. We display all student information and the calculated average.
